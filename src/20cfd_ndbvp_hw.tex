\begin{document}
Consider 2-d PDE
 $$\frac 1 2 \Delta  v(x) - v(x)
  +  x_{1}^{2} +  x_{2}^{2} -  x_{1} -  x_{2} - \frac 3 2 = 0, \ x\in O = (0,1)^{2}$$
with its boundary data
 $$v(x) = (x_{1} - \frac 1 2)^{2} + (x_{2} - \frac 1 2)^{2}, \ x\notin O.$$

 \begin{itemize}
 \item Show that exact solution is
$$v(x) = (x_{1} - \frac 1 2)^{2} + (x_{2} - \frac 1 2)^{2}.$$
\item 
Identify $\gamma, \ell^{h}, p^{h}$ in the CFD solution given by
$$
v (x) = \gamma 
\Big\{ \ell^{h}(x) + 
\sum_{i=1}^{d} 
p^{h}(x+he_{i}|x) v(x+he_{i})
+ p^{h}(x-he_{i}|x) v(x-he_{i})
\Big\}.
$$
\item For $\{h = 2^{-i}, i = 2, 3, 4, 5\}$, compute CFD solution and find maxnorm of error.
\end{itemize}
